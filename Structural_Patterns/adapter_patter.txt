6ï¸âƒ£ Simple Java Example (Basic)
Scenario:

Client expects PaymentProcessor

Step 1: Target Interface
interface PaymentProcessor {
    void pay(int amount);

    void makePayment(double money) {
        System.out.println("Paid via Paytm: " + money);
    }

Step 3: Adapter

    private PaytmGateway paytm;

    PaytmAdapter(PaytmGateway paytm) {
        this.paytm = paytm;
    }

    public void pay(int amount) {
        paytm.makePayment(amount);
    }
}

Step 4: Client
public class Main {
    public static void main(String[] args) {
        PaymentProcessor processor =
            new PaytmAdapter(new PaytmGateway());

        processor.pay(500);
    }
}
